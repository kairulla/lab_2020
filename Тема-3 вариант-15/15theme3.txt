/* 15) Задана матрица А(4,4). Если максимальный элемент матрицы
 *     равен сумме элементов первой строки, то поменять местами
 *     первую строку с той строкой, где находится максимальный элемент. */
#include <cstdlib>
#include<iostream>
#include<math.h>
#include<stdio.h>
#include<string>

//using namespace std;

int main() {
    int a1 = 0;
    int a2 = 0;
    int summa = 0;
    int array[5][5] = {//объявление двух мерного массива и его инициализация 
        {2, 1, 5, 4, 2},
        {5, 6, 4, 7, 5},
        {6, 2, 3, 4, 8},
        {2, 1, 1, 6, 4},
        {5, 6, 9, 7, 1}
    };

    std::cout << "Исходный массив" << std::endl;
    for (int i = 0; i < 5; i++) { //выводим исходный массив
        for (int j = 0; j < 5; j++) {
            std::cout << array[i][j] << " ";
        }
        std::cout << std::endl;
    }
    std::cout << std::endl;

    int max = array[0][0]; //ищем максимальное число в массиве

    for (int i = 0; i < 5; i++) {
        for (int j = 0; j < 5; j++) {
            if (array[i][j] > max) {
                max = array[i][j];
                a2 = i; /* здесь мы узнаёт намер строки максимального элемента
                         * и присваиваем её в переменную a2   */

            }
        }
    }

    for (int j = 0; j < 5; j++) {//сумма элементов первой строки присваеваем в summa

        summa += array[0][j];
    }


    if (a2 == summa) {//проверяется условие равенства максимального элемента и суммы первой строки матрицы
        for (int j = 0; j < 5; j++) {// т.е. если равно, то условие выполняется выполняется
            int b = array[a1][j];
            array[a1][j] = array[a2][j];// этот цикл переворачивает строку, где максимальный элемент, с первой строкой
            array[a2][j] = b;
        }
        for (int i = 0; i < 5; i++) {
            for (int j = 0; j < 5; j++) {
                std::cout << array[i][j] << " ";
            }
            std::cout << '\n';
        }
    } else {//взаимодействие с пользователем при неравенстве
        std::cout << std::endl << std::endl << "Не соответствует условию задачи," << std::endl
                << "потому что " << "сумма первой строки = " << summa
                << ", а наибольший элемент = " << max;
    }
    return 0;
}
